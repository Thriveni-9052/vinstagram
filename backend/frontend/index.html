let API_URL = "http://127.0.0.1:5000"; // 📤 Upload Post function uploadPost() { let image = document.getElementById("image").files[0]; let caption = document.getElementById("caption").value; let formData = new FormData(); formData.append("image", image); formData.append("caption", caption); fetch(`${API_URL}/upload`, { method: "POST", body: formData }) .then(res => res.json()) .then(data => { alert(data.message); location.reload(); }); } // ❤️ Like Post function likePost(postId) { fetch(`${API_URL}/like/${postId}`, { method: "POST" }) .then(res => res.json()) .then(data => { alert("Likes: " + data.likes); location.reload(); }); } // 💬 Comment Post function commentPost(postId) { let comment = prompt("Enter your comment:"); if (comment) { fetch(`${API_URL}/comment/${postId}`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ comment: comment }) }) .then(res => res.json()) .then(data => { alert(data.message); }); } } // 📸 Load Feed function loadFeed() { fetch(`${API_URL}/feed`) .then(res => res.json()) .then(data => { let feed = document.getElementById("feed"); feed.innerHTML = ""; data.forEach(post => { let div = document.createElement("div"); div.className = "post"; div.innerHTML = `

${post.caption}

`; feed.appendChild(div); }); }); } // 🔄 Load feed when page loads window.onload = loadFeed;
